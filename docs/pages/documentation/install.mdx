# Install

```bash copy
npm install glazejs gsap
```

## Setup

```js copy
import gsap from "gsap";
import glaze from "glazejs";

glaze({
  lib: {
    gsap: {
      core: gsap,
    },
  },
});
```

## Config

```js copy
glaze({
  // the root element which will be searched for elements
  element: document,
  // the attribute to search for
  dataAttribute: "data-animate",
  // custom breakpoints
  breakpoints: {
    default: "(min-width: 1px)",
  },
});
```
